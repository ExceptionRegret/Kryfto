# Railway multi-service deployment template

# Usage: Import this repo to Railway, it will auto-detect services.

# Or deploy manually with `railway up` in each service directory.

#

# Required Railway services:

# 1. API — web service (this repo, uses railway.json)

# 2. Worker — background worker

# 3. Redis — Railway plugin or Redis add-on

# 4. Postgres — Railway plugin or Postgres add-on

# 5. MinIO — external S3 bucket or Railway volume

#

# Environment variables to set in Railway dashboard:

#

# Shared:

# POSTGRES_DB=collector

# POSTGRES_USER=collector

# POSTGRES_PASSWORD=<auto-generated>

# POSTGRES_HOST=${{Postgres.PGHOST}}

# POSTGRES_PORT=${{Postgres.PGPORT}}

# REDIS_HOST=${{Redis.REDIS_HOST}}

# REDIS_PORT=${{Redis.REDIS_PORT}}

# KRYFTO_API_TOKEN=<your-token>

# KRYFTO_BOOTSTRAP_ADMIN_TOKEN=<your-token>

# KRYFTO_PROJECT_ID=default

#

# API-specific:

# KRYFTO_PORT=8080

# PORT=8080

# KRYFTO_ARTIFACT_BACKEND=s3

# S3_ENDPOINT=<your-s3-endpoint>

# S3_ACCESS_KEY=<key>

# S3_SECRET_KEY=<secret>

# S3_BUCKET=collector-artifacts

# S3_FORCE_PATH_STYLE=true

#

# Worker-specific:

# KRYFTO_STEALTH_MODE=true

# KRYFTO_ROTATE_USER_AGENT=true

# KRYFTO_PROXY_URLS=<optional>
